import pandas as pd
import matplotlib.pyplot as plt

departments = ["Finance"] * 18 + ["HR"] * 20 + ["Marketing"] * 20 + ["R&D"] * 21 + ["Operations"] * 21
employee_id = [f"EMP{i:03d}" for i in range(1, 101)]
df = pd.DataFrame({"employee_id": employee_id, "department": departments})

finance_count = (df["department"] == "Finance").sum()
print("Finance Department Count:", finance_count)

plt.figure()
df["department"].hist()
plt.show()
